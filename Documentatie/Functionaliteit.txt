CREATE TABLE user_toeslagen (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  user_id INT UNSIGNED NOT NULL,
  toeslag_id INT UNSIGNED NOT NULL,
  actief TINYINT(1) NOT NULL DEFAULT 0,
  bedrag DECIMAL(10,2) DEFAULT NULL,
  jaar YEAR DEFAULT NULL,
  aangemaakt_op TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  bijgewerkt_op TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  CONSTRAINT fk_user_toeslagen_user
    FOREIGN KEY (user_id) REFERENCES users(id)
    ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT fk_user_toeslagen_toeslag
    FOREIGN KEY (toeslag_id) REFERENCES toeslagen(id)
    ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
